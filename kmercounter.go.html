1    package main

3    import (
4    	"bufio"
5    	"fmt"
6    	"io/ioutil"
7    	"log"
8    	"os"
9    	"path/filepath"
10   	"regexp"
11   	"sort"
12   	"strconv"
13   	"strings"
14   )
15   
16   func main() {
17   
18   	argsWithProg := os.Args
19   	argsWithoutProg := os.Args[1:]
20   	fmt.Println(argsWithProg)
21   	fmt.Println(argsWithoutProg)
22   
23   	if (len(os.Args) <2) {
24   		log.Fatalf("This code takes 2 arguments: path to directory and kmer counter")
25   	}
26   	var kmerlength int
27   	kmerlength, err := strconv.Atoi(os.Args[2])
28   
29   	if kmerlength < 0 {
30   		log.Fatalf("Kmer length cannot be less than zero")
31   	} else {
32   		fmt.Println("Desired kmer length is %s", kmerlength)
33   	}
34   
35   	files, err := ioutil.ReadDir(os.Args[1])
36   	if err != nil {
37   		log.Fatal(err)
38   	}
39   	var path string
40   	path, err = os.Getwd()
41   	if err != nil {
42   		log.Fatal(err)
43   	}
44   
45   	for _, f := range files {
46   
47   		if strings.HasSuffix(f.Name(), "fasta") {
48   			file, err := os.Open(f.Name())
49   			if err != nil {
50   				log.Fatal(err)
51   			} else
52   			{
53   				fmt.Println(filepath.Join(path, f.Name()))
54   			}
55   			defer file.Close()
56   			scanner := bufio.NewScanner(file)
57   			scanner.Split(bufio.ScanLines)
58   			var (
59   				text      []string
60   				totaltext string
61   			)
62   
63   			var linenumber int
64   			var linereadin string
65   			linenumber = 1
66   			for scanner.Scan() {
67   				linereadin = strings.ToLower(scanner.Text())
68   				if linenumber > 1 {
69   				text = append(text, linereadin)
70   				totaltext = totaltext + strings.ToLower(scanner.Text())
71   				}
72   				linenumber++
73   			}
74   
75   			// The method os.File.Close() is called
76   			// on the os.File object to close the file
77   			file.Close()
78   
79   			totaltext = strings.ReplaceAll(totaltext, ">", "")
80   			fmt.Println(totaltext)
81   			// check if the fasta file contains only A T G C
82   			isDNA := regexp.MustCompile(`^[atgc]`).MatchString
83   
84   			if !isDNA(totaltext) {
85   				fmt.Printf("%q is not valid\n", f.Name())
86   				log.Fatalf("Not valid DNA file")
87   			} else {
88   				fmt.Println("Valid Fasta File")
89   			}
90   
91   			var (
92   				kmerlengths = make(map[string]int, 1000)
93   			)
94   
95   			fmt.Printf("Length of sequence in file %s is %d\n", f.Name(), len(text))
96   			// if length is greater than zero, then count the kimer length
97   			// check if kmer length is greater than length of file
98   			if kmerlength > len(text) {
99   				log.Fatalf("Kmer length cannot be greater than file length")
100  			}
101  			if len(text) > 0 {
102  				for i := 0; i < len(totaltext)-kmerlength; i = i + kmerlength {
103  					var _, checkkey = kmerlengths[totaltext[i:i+kmerlength]]
104  					if checkkey {
105  						kmerlengths[totaltext[i:i+kmerlength]]++
106  					} else {
107  						kmerlengths[totaltext[i:i+kmerlength]] = 1
108  					}
109  				}
110  			} else
111  			{
112  				log.Fatalf("Empty fasta file")
113  			}
114  			type kv struct {
115  				Key   string
116  				Value int
117  			}
118  			var ss []kv
119  			for k, v := range kmerlengths {
120  				ss = append(ss, kv{k, v})
121  			}
122  
123  			sort.Slice(ss, func(i, j int) bool {
124  				return ss[i].Value > ss[j].Value
125  			})
126  
127  
128  			//Write it to file
129  			opfile, err := os.Create(f.Name()+"."+os.Args[2]+".counts.txt")
130  
131  			if (err != nil) {
132  				log.Fatalf("Unable to open output file %s", f.Name())
133  			} else {
134  				fmt.Printf("Writing to file %s", opfile.Name())
135  			}
136  
137  			defer opfile.Close()
138  
139  			for _, kv := range ss {
140  				fmt.Println("Key:", kv.Key, "Value:", kv.Value)
141  				_, err2 := opfile.WriteString(
142  					"Kmer: "+
143  					kv.Key+
144  					"Count: "+
145  					string(kv.Value)+
146  				 "\n")
147  
148  				if err2 != nil {
149  					log.Fatal(err2)
150  				}
151  
152  			}
153  			fmt.Printf("Finished writing to file %s ", opfile.Name())
154  		}
155  
156  	}
157  
158  	if err != nil {
159  		fmt.Println("File reading error", err)
160  		return
161  	}
162  	//fmt.Println("Contents of file:", string(data))
163  }
